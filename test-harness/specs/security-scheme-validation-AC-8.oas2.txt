====test====
Resource is returned when an api-key is provided as a header.
====spec====
swagger: '2.0'
paths:
  /todos:
    get:
      produces:
        - application/json
      responses:
        200:
          description: OK
          examples:
            application/json: test
securityDefinitions:
  ApiKeyAuth:
    type: apiKey
    in: header
    name: X-API-Key
security:
  - ApiKeyAuth: []
====server====
mock -p 4010
====command====
curl -i http://localhost:4010/todos -H "X-API-Key: abc123"
====expect====
HTTP/1.1 200 OK

"test"
